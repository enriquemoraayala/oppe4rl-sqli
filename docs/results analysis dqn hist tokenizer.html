    total_score=pd.NamedAgg(column="reward", aggfunc="sum"),
    win=pd.NamedAgg(column="win", aggfunc="sum")

import matplotlib.pyplot as plt

fig, ax = plt.subplots(figsize=(16, 9))
df_last = df.sort_values(["eps_id", "mutation_num"]).drop_duplicates(["eps_id"], keep="last")

df_last["win_cummean"] = df_last["win"].expanding().mean()
df_last.plot(x="eps_id", y="win_cummean", ax=ax, label="global", ylabel="Cummulative win proportion", xlabel="Episode");

for original_payload in df_last["original_payload"].unique():
    df_last_p = df_last[df_last["original_payload"] == original_payload]
    df_last_p["win_cummean"] = df_last_p["win"].expanding().mean()
    df_last_p = df_last_p.set_index("eps_id").reindex(df_last.eps_id, method="ffill", fill_value=0).reset_index()
    df_last_p.plot(x="eps_id", y="win_cummean", ax=ax, label=original_payload[:20], ylabel="Cummulative win proportion", xlabel="Episode")
